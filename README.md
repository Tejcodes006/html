set a 2

declare 
rn stud.rno%TYPE;
name stud.sname%TYPE;
cl stud.class%TYPE;
p stud.per %TYPE;
gr stud.grade%TYPE;
begin
rn:=&rn;
name:='&name';
cl:='&cl';
p:=&p;
IF(P>75)THEN
gr:='disitinction';
ELSIF((P>60)AND(P<75))THEN
gr:='first class';
ELSIF((P>50)AND(P<60))THEN
gr:='second class';
ELSIF((P>=35)AND(P<45))THEN
gr:='passed';
ELSE
gr:='failed';
END IF;
insert into stud values(rn,name,cl,p,gr);
dbms_output.put_line('insert record successfully');
end;
